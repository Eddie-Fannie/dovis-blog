# 349.两个数组的交集
> 给定两个数组，编写一个函数来计算它们的交集。

## 示例
```bash
输入：nums=[1,2,2,1], nums2=[2,2]
输出；[2]

输入：nums1 = [4,9,5], nums=[9,4,9,8,4]
输出：[9,4]
```
- 输出结果中的每个元素一定是唯一的
- 我们可以不考虑输出结果的顺序

## 自己的答案
```js
/**
 * @param {number[]} nums1
 * @param {number[]} nums2
 * @return {number[]}
 */
var intersection = function(nums1, nums2) {
    if(nums1.length ===0 || nums2.length === 0) return [];
    let newNums1 = deleteSame(nums1);
    let newNums2 = deleteSame(nums2);
    let result = [];
    for(let j =0;j<newNums2.length;j++) {
        if(newNums1.indexOf(newNums2[j]) > -1) {
            result.push(newNums2[j])
        }
    }
    return result;
};
var deleteSame = function(num) {
    let newArr = [];
    let newMap = new Map()
    for(let i in num) {
        if(newMap.has(num[i])) {
            newMap.set(num[i], true)
        } else {
            newMap.set(num[i], false)
            newArr.push(num[i])
        }
    }
    return newArr;
}
```

## 别人的答案
```js

/**
 * @param {number[]} nums1
 * @param {number[]} nums2
 * @return {number[]}
 */

var intersection = function(nums1, nums2) {
    let arr = [];
    for(let i = 0; i < nums1.length; i++) {
        if (nums2.includes(nums1[i])) {
            arr.push(nums1[i]);
        }
    }
    return Array.from(new Set(arr))
};
```
